#tcp
sudo iptables -t nat -A PREROUTING -p tcp --dport 25565 -j DNAT --to-destination 10.160.196.2:25565
sudo iptables -A FORWARD -i eth0 -o wg0 -p tcp -m tcp --dport 25565 -j ACCEPT

#udp
sudo iptables -t nat -A PREROUTING -p udp --dport 25565 -j DNAT --to-destination 10.160.196.2:25565
sudo iptables -A FORWARD -i eth0 -o wg0 -p udp -m udp --dport 25565 -j ACCEPT
